def troca(l):
    lista1 = []
    lista2 = []

    troca = 0
    while troca < len(l)/2:
        lista1.append(l[troca])
        troca += 1
    troca2 = 0
    while troca2 < len(l):
        lista2.append(l[troca2])
        troca2 += 1
        troca2 = int(len(l)/2)
        return troca2
